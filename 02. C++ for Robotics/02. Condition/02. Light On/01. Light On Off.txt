#define LED_WHITE 10
#define LED_RED 8

bool white = true;

void setup() {
  pinMode(LED_WHITE, OUTPUT);
  pinMode(LED_RED, OUTPUT);
}

void loop() {
  if(white)
  {
    digitalWrite(LED_WHITE, HIGH);
    digitalWrite(LED_RED, LOW);
    delay(500);
    white = false;
  }
  else
  {
    digitalWrite(LED_WHITE, LOW);
    digitalWrite(LED_RED, HIGH);
    delay(500);
    white = true;
  }
}
