#include <Servo.h>

Servo myServo; 
long exc_time = 0;
long cur_time = 0;

void setup() {
  Serial.begin(9600);
  myServo.attach(9);      // pwm digital pin

  pinMode(A0, INPUT);
  
}

void loop() {
  int irValue = analogRead(A0);
  Serial.println(irValue);

  if(irValue < 100)
  {
    myServo.write(180);
    exc_time = millis();

  }
  else
  {
    cur_time = millis();
    if(cur_time - exc_time >= 1500)
    {
      myServo.write(0);
    }
  }

}
