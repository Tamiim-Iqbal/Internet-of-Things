#define LDR A0

void setup() {
  Serial.begin(9600);
  pinMode(LDR, INPUT);
  pinMode(3, OUTPUT);
  pinMode(4, OUTPUT);
  pinMode(5, OUTPUT);
  pinMode(6, OUTPUT);
  pinMode(7, OUTPUT);

}

void loop() {
  int LDRvalue = analogRead(LDR);
  Serial.println(LDRvalue);

  if(LDRvalue <= 300)
  {
    for (int pin = 3; pin <= 7; pin++)
    {
      digitalWrite(pin, HIGH);
    }
  }
  else
  {
    for (int pin = 3; pin <= 7; pin++)
    {
      digitalWrite(pin, LOW);
    }
  }

  delay(500);
}
