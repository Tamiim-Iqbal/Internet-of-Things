#define buttonPin 8
#define ledPin 6
int brightness = 0;

void setup() {
  Serial.begin(9600);
  pinMode(buttonPin, INPUT_PULLUP);
  pinMode(ledPin, OUTPUT);
}

void loop() {
  int pinState = digitalRead(buttonPin);

  if(pinState == LOW) {
    delay(1000);

    brightness += 50;

    if(brightness > 255) {
      brightness = 0;                 // Reset brightness to 0 if it exceeds 255
    }

    analogWrite(ledPin, brightness);  // Use analogWrite to control brightness
  }
}
